#!/bin/bash

set -euo pipefail

# Flags CUDA + C++23
CXXFLAGS="-std=c++23 -O3 -Wall -Wextra -Wpedantic"
CUDAFLAGS="-arch=sm_92"  # RTX 5060 Ti (Blackwell)

SRC="kernel.cu"

echo "Compilation CUDA OK."

#!/bin/bash

set -euo pipefail

CXXFLAGS="-std=c++23 -O3 -Wall -Wextra -Wpedantic"
CUDAFLAGS="-arch=sm_92"

SRC=(
    main.cpp
    kernel.cu
    utils.cpp
)

OUT="program"

nvcc $CXXFLAGS $CUDAFLAGS "${SRC[@]}" -o "$OUT"

echo "Compilation CUDA OK."
./"$OUT"
